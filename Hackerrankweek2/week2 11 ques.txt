#include <stdio.h>
#include <string.h>

int superDigitHelper(long long num) {
    if (num < 10) {
        return (int)num; 
    }

    long long sum = 0;
    while (num > 0) {
        sum += num % 10;
        num /= 10;
    }
    return superDigitHelper(sum);
}

int superDigit(char n[], int k) {
    long long sumN = 0;
    for (int i = 0; n[i] != '\0'; i++) {
        sumN += (n[i] - '0');
    }

    long long initial = sumN * k;

    return superDigitHelper(initial);
}

int main() {
    char n[100001];
    int k;
    scanf("%s %d", n, &k);

    printf("%d\n", superDigit(n, k));
    return 0;
}
