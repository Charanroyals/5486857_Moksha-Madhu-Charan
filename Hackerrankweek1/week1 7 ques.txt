#include <stdio.h>
#include <stdlib.h>  

int diagonalDifference(int n, int matrix[n][n]) {
    int sumPrimary = 0;
    int sumSecondary = 0;

    for (int i = 0; i < n; i++) {
        sumPrimary += matrix[i][i];
        sumSecondary += matrix[i][n - i - 1];
    }

    return abs(sumPrimary - sumSecondary);
}

int main() {
    int n;
    scanf("%d", &n);

    int matrix[n][n];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            scanf("%d", &matrix[i][j]);
        }
    }

    int result = diagonalDifference(n, matrix);
    printf("%d\n", result);

    return 0;
}
