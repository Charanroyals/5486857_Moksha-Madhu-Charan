#include <stdio.h>

int lonelyinteger(int arr[], int n) {
    int unique = 0;
    for (int i = 0; i < n; i++) {
        unique ^= arr[i]; 
    }
    return unique;
}

int main() {
    int n;
    scanf("%d", &n);

    int numbers[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &numbers[i]);
    }

    int result = lonelyinteger(numbers, n);
    printf("%d\n", result);

    return 0;
}
