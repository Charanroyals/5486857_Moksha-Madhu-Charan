#include <stdio.h>
#include <string.h>
#include <stdlib.h>

void timeConversion(char *s) {
    int hour;
    char period[3];
    char result[9];  

    sscanf(s, "%2d:%*2d:%*2d%2s", &hour, period);

    if (strcmp(period, "AM") == 0) {
        if (hour == 12) {
            hour = 0;
        }
    } else { // PM case
        if (hour != 12) {
            hour += 12;
        }
    }

    snprintf(result, sizeof(result), "%02d:%c%c:%c%c",
             hour, s[3], s[4], s[6], s[7]);

    printf("%s\n", result);
}

int main() {
    char timeString[11];
    scanf("%s", timeString);

    timeConversion(timeString);

    return 0;
}
