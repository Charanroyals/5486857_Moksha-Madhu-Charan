#include <stdio.h>

#define MAX_VALUE 100 

void countingSort(int arr[], int n) {
    int frequency[MAX_VALUE] = {0}; 
    
    for (int i = 0; i < n; i++) {
        frequency[arr[i]]++;
    }


    for (int i = 0; i < MAX_VALUE; i++) {
        printf("%d ", frequency[i]);
    }
    printf("\n");
}

int main() {
    int n;
    scanf("%d", &n);
    
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    countingSort(arr, n);

    return 0;
}
