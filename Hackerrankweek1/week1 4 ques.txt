#include <stdio.h>
#include <string.h>

void matchingStrings(char words[][101], int n, char queries[][101], int q) {
    for (int i = 0; i < q; i++) {
        int count = 0;
        for (int j = 0; j < n; j++) {
            if (strcmp(words[j], queries[i]) == 0) {
                count++;
            }
        }
        printf("%d\n", count);
    }
}

int main() {
    int n;
    scanf("%d", &n);

    char words[n][101]; 
    for (int i = 0; i < n; i++) {
        scanf("%s", words[i]);
    }

    int q;
    scanf("%d", &q);

    char queries[q][101];
    for (int i = 0; i < q; i++) {
        scanf("%s", queries[i]);
    }

    matchingStrings(words, n, queries, q);

    return 0;
}
