#include <stdio.h>
#include <stdint.h>   // for uint32_t

unsigned int flippingBits(unsigned int n) {
    return ~n & 0xFFFFFFFF;
}

int main() {
    int q;
    scanf("%d", &q);

    while (q--) {
        unsigned int num;
        scanf("%u", &num);

        printf("%u\n", flippingBits(num));
    }

    return 0;
}
