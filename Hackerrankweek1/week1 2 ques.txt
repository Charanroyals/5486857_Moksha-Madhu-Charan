#include <stdio.h>

void miniMaxSum(long long values[], int length) {
    long long total = 0;
    long long smallest = values[0];
    long long largest = values[0];

    for (int i = 0; i < length; i++) {
        total += values[i];
        if (values[i] < smallest) {
            smallest = values[i];
        }
        if (values[i] > largest) {
            largest = values[i];
        }
    }

    long long minSum = total - largest;
    long long maxSum = total - smallest;

    printf("%lld %lld\n", minSum, maxSum);
}

int main() {
    long long arr[5];
    for (int i = 0; i < 5; i++) {
        scanf("%lld", &arr[i]);
    }

    miniMaxSum(arr, 5);

    return 0;
}
