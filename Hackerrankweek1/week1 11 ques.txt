#include <stdio.h>

int birthday(int squares[], int n, int d, int m) {
    int count = 0;

    for (int i = 0; i <= n - m; i++) {
        int sum = 0;
        for (int j = 0; j < m; j++) {
            sum += squares[i + j];
        }
        if (sum == d) {
            count++;
        }
    }

    return count;
}

int main() {
    int n;
    scanf("%d", &n);

    int squares[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &squares[i]);
    }

    int d, m;
    scanf("%d %d", &d, &m);

    int result = birthday(squares, n, d, m);
    printf("%d\n", result);

    return 0;
}
