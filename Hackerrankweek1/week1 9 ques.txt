#include <stdio.h>
#include <string.h>
#include <ctype.h>

const char* pangrams(char *s) {
    int seen[26] = {0}; 
    int uniqueCount = 0;

    for (int i = 0; s[i] != '\0'; i++) {
        if (isalpha(s[i])) {  
            int index = tolower(s[i]) - 'a';
            if (seen[index] == 0) {
                seen[index] = 1;
                uniqueCount++;
            }
        }
    }

    if (uniqueCount == 26) {
        return "pangram";
    } else {
        return "not pangram";
    }
}

int main() {
    char sentence[1000];
    
    fgets(sentence, sizeof(sentence), stdin);
    
    sentence[strcspn(sentence, "\n")] = '\0';

    printf("%s\n", pangrams(sentence));

    return 0;
}
