#include <stdio.h>

int main() {
    int n, m;
    scanf("%d", &n);

    int ranked[n];
    for (int i = 0; i < n; i++) scanf("%d", &ranked[i]);

    scanf("%d", &m);
    int player[m];
    for (int i = 0; i < m; i++) scanf("%d", &player[i]);

    int unique[n], uCount = 0;
    unique[uCount++] = ranked[0];
    for (int i = 1; i < n; i++) {
        if (ranked[i] != ranked[i-1]) {
            unique[uCount++] = ranked[i];
        }
    }

    int positive = uCount - 1;
    for (int i = 0; i < m; i++) {
        while (positive >= 0 && player[i] >= unique[positive]) {
            positive--;
        }
        printf("%d\n", positive + 2);
    }

    return 0;
}
